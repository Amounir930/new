# 🏗️ خطة مشروع Apex Platform v3.0 (Tier-1 Global Architecture - Zero-Waste)
## النسخة الهندسية المحسنة v3.1 - Atomic Capability Library & Self-Hosted Excellence

تم إعادة هندسة الخطة بالكامل لتتوافق مع معايير البنية التحتية من المستوى الأول (Tier-1) مع الاعتماد على الاستضافة الذاتية (Self-Hosted) للأدوات عالية الأداء، وإضافة **مكتبة القدرات الذرية (Atomic Capability Library)** لتمكين تخصيص المتاجر بالذكاء الاصطناعي بأمان تام.

### 2. `CapabilityMetadata` Interface (Atomic Capability Library)

```typescript
// packages/capabilities/core/src/interfaces/capability-metadata.interface.ts

import { ZodSchema } from 'zod';

/**
 * Capability Metadata Interface
 * Describes what each capability does for AI-Bridge discovery and execution
 */
export interface CapabilityMetadata {
  /** Unique identifier for the capability */
  name: string;
  
  /** Semantic version of the capability */
  version: string;
  
  /** Human-readable description for AI intent matching */
  description: string;
  
  /** List of actions this capability can perform */
  actions: CapabilityAction[];
  
  /** Dependencies on other capabilities */
  dependencies: string[];
  
  /** Whether this capability can be modified by AI */
  aiEnabled: boolean;
  
  /** Whether this capability can modify ThemeConfig */
  themeConfigAccess: boolean;
}

export interface CapabilityAction {
  /** Action name (e.g., "createDiscount", "generateContent") */
  name: string;
  
  /** Description for AI intent matching */
  description: string;
  
  /** Zod schema for input validation */
  inputSchema: ZodSchema;
  
  /** Zod schema for output validation */
  outputSchema: ZodSchema;
  
  /** Required permissions to execute this action */
  requiredPermissions: string[];
}

// packages/capabilities/core/src/interfaces/theme-config.interface.ts

import { z } from 'zod';

/**
 * ThemeConfig Schema - AI can ONLY modify these fields
 * This ensures backend core logic remains immutable (Safe-Core)
 */
export const ThemeConfigSchema = z.object({
  colors: z.object({
    primary: z.string().regex(/^#[0-9A-F]{6}$/i),
    secondary: z.string().regex(/^#[0-9A-F]{6}$/i),
    accent: z.string().regex(/^#[0-9A-F]{6}$/i),
  }),
  fonts: z.object({
    heading: z.string(),
    body: z.string(),
  }),
  layout: z.object({
    headerStyle: z.enum(['default', 'minimal', 'centered']),
    footerStyle: z.enum(['default', 'minimal']),
    productGridColumns: z.number().min(1).max(6),
  }),
  branding: z.object({
    logoUrl: z.string().url().optional(),
    faviconUrl: z.string().url().optional(),
  }),
});

export type ThemeConfig = z.infer<typeof ThemeConfigSchema>;

// packages/capabilities/core/src/bridge/ai-capability-bridge.ts

import { Ollama } from 'ollama';
import { CapabilityRegistry } from '../registry/capability-registry';
import { ThemeConfigSchema } from '../interfaces/theme-config.interface';
import { z } from 'zod';

/**
 * AI-Bridge Middleware
 * Translates natural language to validated Capability Selection List
 */
export class AICapabilityBridge {
  constructor(
    private readonly ollama: Ollama,
    private readonly capabilityRegistry: CapabilityRegistry,
  ) {}

  async translateNaturalLanguageToCapabilities(
    userInput: string,
    tenantId: string,
  ): Promise<CapabilitySelectionList> {
    // Step 1: Get all capability metadata for RAG context
    const capabilities = await this.capabilityRegistry.getAllMetadata();
    
    // Step 2: Use Ollama to extract intent and parameters
    const ollamaResponse = await this.ollama.chat({
      model: 'llama3',
      messages: [{
        role: 'user',
        content: `
          Analyze this user request and map it to available capabilities:
          User: "${userInput}"
          
          Available Capabilities:
          ${JSON.stringify(capabilities, null, 2)}
          
          Return a JSON array of capability selections with validated parameters.
        `,
      }],
      format: 'json',
    });

    // Step 3: Parse and validate against capability schemas
    const selectionList: CapabilitySelectionList = JSON.parse(ollamaResponse.message.content);
    
    // Step 4: Validate each selection against its capability's input schema
    for (const selection of selectionList.selections) {
      const capability = await this.capabilityRegistry.getCapability(selection.capabilityName);
      const action = capability.metadata.actions.find(a => a.name === selection.action);
      
      if (!action) {
        throw new Error(`Invalid action: ${selection.action}`);
      }
      
      // CRITICAL: Validate parameters against Zod schema
      action.inputSchema.parse(selection.parameters);
      
      // CRITICAL: If theme config modification, validate against ThemeConfigSchema
      if (capability.metadata.themeConfigAccess) {
        ThemeConfigSchema.partial().parse(selection.parameters);
      }
    }
    
    // Step 5: Log to audit
    await this.capabilityRegistry.auditExecution(tenantId, selectionList);
    
    return selectionList;
  }
}

export interface CapabilitySelectionList {
  selections: Array<{
    capabilityName: string;
    action: string;
    parameters: Record<string, any>;
  }>;
}
```
